def parallel(prompt: str, inputs: List[str], n_workers: int = 3) -> List[str]:
    """ä½¿ç”¨ç›¸åŒçš„æç¤ºå¹¶å‘å¤„ç†å¤šä¸ªè¾“å…¥ã€‚"""

        return [f.result() for f in futures]
